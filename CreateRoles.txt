--пользователь только для чтения
CREATE ROLE user_read_only WITH
  NOLOGIN
  NOSUPERUSER
  INHERIT
  NOCREATEDB
  NOCREATEROLE
  NOREPLICATION;
GRANT SELECT ON ALL TABLES IN SCHEMA public TO user_read_only;
